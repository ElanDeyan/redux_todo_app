  Expression<int>? id,
  Expression<String>? title,
  Expression<String>? description,
  Expression<bool>? isCompleted,
}) {

    if (id != null) 'id': id,
    if (title != null) 'title': title,
    if (description != null) 'description': description,
    if (isCompleted != null) 'is_completed': isCompleted,
  });
}
